NAME

androHTML/form

FUNCTION

The PHP method form creates an HTML FORM node, adds it as a child to the current node, and returns a reference to it.

All inputs are optional.

INPUTS

string $name - the name of the form. Default: "Form1" string $method - the method (GET or POST). Default: POST string $action - the URI to route to. Default "index.php" string $x6page - the value of x6page to go to, defaults to none.

SOURCE

function &form($name='Form1',$method='POST',$action="index.php",$x6page=''){
    $form = $this->h('form');
    $form->hp['id']     = $name;
    $form->hp['name']   = $name;
    $form->hp['method'] = $method;
    $form->hp['action'] = $action;
    $form->hp['enctype']="multipart/form-data";
    if($x6page <> '') {
        $symbol = strpos($action,'?')===false ? '?' : '&';
        $form->hp['action'].=$symbol."x6page=".$x6page;
    }
    return $form;
}
