NAME

print_r

FUNCTION

The method dumps a variable into the innerHTML of an and Andromeda HTML Object.

SOURCE

function print_r($value) {
    ob_start();
    print_r($value);
    $pre = $this->h('pre',ob_get_clean());
    $pre->hp['class'] = 'border: 1px solid gray; background-color:white;
        color: black;';
}
